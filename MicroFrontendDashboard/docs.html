API Reference

Widget API

Widget Registration

// Register a new widget
window.MicroFE.registerWidget({
  id: 'my-widget',
  version: '1.0.0',
  mount: (container, props) => { /* ... */ },
  unmount: (container) => { /* ... */ },
  update: (container, props) => { /* ... */ }
});

Event Communication

// Emit events to other widgets
window.MicroFE.emit('data-updated', { 
  source: 'my-widget', 
  data: payload 
});

// Listen for events from other widgets
window.MicroFE.on('data-updated', (event) => {
  // Handle the event
});

Data Binding

// Bind to shared data
window.MicroFE.bind('user-selection', (data) => {
  // React to user selection changes
});

// Update shared data
window.MicroFE.update('user-selection', newSelection);
